        server 127.0.0.1:3005;

    server {
        listen 443 ssl; 
        server_name localhost;

        ssl_certificate      "C:/Users/me505/Desktop/devops/nginx-certs/nginx-selfsigned.crt";
        ssl_certificate_key  "C:/Users/me505/Desktop/devops/nginx-certs/nginx-selfsigned.key";

        location / {
            proxy_pass http://nodejs_cluster;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
        }
    }
    }
    server {
        listen 8080; 
        server_name localhost;

        location / {
            return 301 https://$host$request.uri;
        }
    }
